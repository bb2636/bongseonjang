import { getApiUrl } from '../../config/api.js';

const API_BASE_URL = getApiUrl('/api/payments');

const getAuthHeaders = () => {
  const token = localStorage.getItem('fitting_auth_token');
  return {
    'Content-Type': 'application/json',
    ...(token && { Authorization: `Bearer ${token}` })
  };
};

const PaymentService = {
  async getMatchingPaymentParams(productId) {
    const callbackUrl = `${window.location.origin}/api/payments/nicepay/callback`;

    const response = await fetch(`${API_BASE_URL}/checkout/matching`, {
      method: 'POST',
      headers: getAuthHeaders(),
      body: JSON.stringify({ productId, returnUrl: callbackUrl })
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || '결제 정보 생성에 실패했습니다.');
    }

    return response.json();
  },

  async getPtPaymentParams(productId) {
    const callbackUrl = `${window.location.origin}/api/payments/nicepay/callback`;

    const response = await fetch(`${API_BASE_URL}/checkout/pt`, {
      method: 'POST',
      headers: getAuthHeaders(),
      body: JSON.stringify({ productId, returnUrl: callbackUrl })
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || '결제 정보 생성에 실패했습니다.');
    }

    return response.json();
  },

  async getPaymentResult(orderId) {
    const response = await fetch(`${API_BASE_URL}/result?orderId=${orderId}`, {
      method: 'GET',
      headers: getAuthHeaders()
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || '결제 결과 조회에 실패했습니다.');
    }

    return response.json();
  },

  async getPaymentHistory() {
    const response = await fetch(`${API_BASE_URL}/history`, {
      method: 'GET',
      headers: getAuthHeaders()
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || '결제 내역 조회에 실패했습니다.');
    }

    return response.json();
  }
};

export default PaymentService;
