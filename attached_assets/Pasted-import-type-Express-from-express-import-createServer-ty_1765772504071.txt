const KAKAO_REST_API_KEY = process.env.KAKAO_REST_API_KEY;
const KAKAO_JAVASCRIPT_KEY = process.env.KAKAO_JAVASCRIPT_KEY;
const KAKAO_ADDRESS_SEARCH_URL = "https://dapi.kakao.com/v2/local/search/address.json";
const NAVER_CLIENT_ID = process.env.NAVER_CLIENT_ID;

const upload = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: 10 * 1024 * 1024,
  },
  fileFilter: (req, file, cb) => {
    const allowedTypes = ["image/jpeg", "image/png", "image/gif", "image/webp"];
    if (allowedTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error("Invalid file type"));
    }
  },
});

const storageClient = new Client();

export async function registerRoutes(app: Express): Promise<Server> {
  app.use("/api/auth", authRoutes);
  app.use("/api/auth", onboardingRoutes);
  app.use("/api/posts", postRoutes);
  app.use("/api/home", homeRoutes);
  app.use("/api/users", profileRoutes);
  app.use("/api/user", userRoutes);
  app.use("/api/pt-profile", ptProfileRoutes);
  app.use("/api/blocks", blockRoutes);
  app.use("/api/reports", reportRoutes);
  app.use("/api/profile", profileEditRoutes);
  app.use("/api/chat", chatRoutes);
  app.use("/api/banners", bannerRoutes);
  app.use("/api/referral", referralRoutes);
  app.use("/api/ads", adRoutes);
  app.use("/api/products", productRoutes);
  app.use("/api/payments", paymentRoutes);
  app.use("/api/sms", smsRoutes);
  app.use("/api/admin", adminAuthRoutes);
  app.use("/api/admin", adminUserRoutes);
  app.use("/api/admin/reports", adminReportRoutes);
  app.use("/api/admin/payments", adminPaymentRoutes);
  app.use("/api/admin/products", adminProductRoutes);
  app.use("/api/main-banners", mainBannerRoutes);
  app.use("/api/sub-banners", subBannerRoutes);
  app.use("/api/partnerships", partnershipRoutes);
  app.use("/api/profile-view", profileViewRoutes);
  app.use("/api/notifications", notificationRoutes);
  app.use("/api/likes", likeRoutes);
  app.use("/api/attendance", attendanceRoutes);
  app.use("/api/admin/pt-profiles", adminPTProfileRoutes);
  app.use("/api/pt-application", ptApplicationRoutes);
  app.use("/api/admin/pt-applications", adminPtApplicationRoutes);
  app.use("/api/email", emailRoutes);
  app.use("/api/policies", policyRoutes);
  app.use("/api/admin/policies", adminPolicyRoutes);

  app.post("/api/upload/post-image", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const postId = req.body.postId || randomUUID();
      const imageType = req.body.imageType || "additional";
      const imageIndex = req.body.imageIndex || randomUUID();
      
      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      
      let objectName: string;
      if (imageType === "main") {
        objectName = `post-images/${postId}/main.${fileExtension}`;
      } else {
        objectName = `post-images/${postId}/additional/${imageIndex}.${fileExtension}`;
      }

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
        postId,
        imageType,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.post("/api/upload/certificate-image", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `certificate-images/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Internal server error" });
    }
  });

  app.post("/api/upload/affiliation-image", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `affiliation-images/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.post("/api/upload/body-photo", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `pt-body/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.post("/api/upload/profile", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `user-profiles/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.post("/api/upload/partnership-image", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `partnership-images/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.post("/api/upload/banner-image", upload.single("image"), async (req, res) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: "No file uploaded" });
      }

      const fileExtension = req.file.originalname.split(".").pop() || "jpg";
      const objectName = `banner-images/${randomUUID()}.${fileExtension}`;

      const result = await storageClient.uploadFromBytes(
        objectName,
        req.file.buffer,
        { compress: false }
      );

      if (!result.ok) {
        console.error("Upload error:", result.error);
        return res.status(500).json({ error: "Failed to upload file" });
      }

      res.json({
        success: true,
        objectPath: `/objects/${objectName}`,
      });
    } catch (error) {
      console.error("Upload error:", error);
      res.status(500).json({ error: "Failed to upload file" });
    }
  });

  app.get(/^\/objects\/(.+)$/, async (req, res) => {
    try {
      const objectPath = req.params[0];

      const existsResult = await storageClient.exists(objectPath);
      if (!existsResult.ok || !existsResult.value) {
        return res.status(404).json({ error: "File not found" });
      }

      const stream = storageClient.downloadAsStream(objectPath, {
        decompress: false,
      });

      const extension = objectPath.split(".").pop()?.toLowerCase();
      const mimeTypes: Record<string, string> = {
        jpg: "image/jpeg",
        jpeg: "image/jpeg",
        png: "image/png",
        gif: "image/gif",
        webp: "image/webp",
      };

      res.set({
        "Content-Type": mimeTypes[extension || ""] || "application/octet-stream",
        "Cache-Control": "public, max-age=31536000",
      });

      stream.pipe(res);
    } catch (error) {
      console.error("Download error:", error);
      res.status(500).json({ error: "Failed to download file" });
    }
  });

  app.delete(/^\/api\/objects\/(.+)$/, async (req, res) => {
    try {
      const objectPath = req.params[0];

      const result = await storageClient.delete(objectPath, {
        ignoreNotFound: true,
      });

      if (!result.ok) {
        console.error("Delete error:", result.error);
        return res.status(500).json({ error: "Failed to delete file" });
      }

      res.json({ success: true });
    } catch (error) {
      console.error("Delete error:", error);
      res.status(500).json({ error: "Failed to delete file" });
    }
  });

  app.get("/api/health", (req, res) => {
    res.json({ status: "ok", timestamp: new Date().toISOString() });
  });

  app.get("/api/config/kakao", (req, res) => {
    if (!KAKAO_JAVASCRIPT_KEY) {
      return res.status(500).json({ 
        success: false, 
        error: "카카오 JavaScript 키가 설정되지 않았습니다." 
      });
    }
    if (!KAKAO_REST_API_KEY) {
      return res.status(500).json({ 
        success: false, 
        error: "카카오 REST API 키가 설정되지 않았습니다." 
      });
    }
    res.json({ 
      success: true, 
      javascriptKey: KAKAO_JAVASCRIPT_KEY,
      restApiKey: KAKAO_REST_API_KEY
    });
  });

  app.get("/api/config/naver", (req, res) => {
    if (!NAVER_CLIENT_ID) {
      return res.status(500).json({ 
        success: false, 
        error: "네이버 Client ID가 설정되지 않았습니다." 
      });
    }
    res.json({ 
      success: true, 
      clientId: NAVER_CLIENT_ID 
    });
  });

  app.get("/api/location/search", async (req, res) => {
    try {
      const query = req.query.query as string;

      if (!query || query.trim() === "") {
        return res.json({ success: true, data: [] });
      }

      if (!KAKAO_REST_API_KEY) {
        console.error("KAKAO_REST_API_KEY is not configured");
        return res.status(500).json({ 
          success: false, 
          error: "API 키가 설정되지 않았습니다." 
        });
      }

      const params = new URLSearchParams({
        query: query.trim(),
        size: "15",
      });

      const apiUrl = `${KAKAO_ADDRESS_SEARCH_URL}?${params}`;
      console.log("Calling Kakao API:", apiUrl);

      const response = await fetch(apiUrl, {
        method: "GET",
        headers: {
          "Authorization": `KakaoAK ${KAKAO_REST_API_KEY}`,
        },
      });

      console.log("Kakao API response status:", response.status);

      if (response.status === 401) {
        return res.status(401).json({ 
          success: false, 
          error: "API 키가 유효하지 않습니다." 
        });
      }

      if (response.status === 403) {
        return res.status(403).json({ 
          success: false, 
          error: "카카오맵 API 권한이 필요합니다. 카카오 개발자 콘솔에서 카카오맵 사용 설정을 ON으로 변경해주세요." 
        });
      }

      if (response.status === 429) {
        return res.status(429).json({ 
          success: false, 
          error: "일일 사용량을 초과했습니다." 
        });
      }

      if (!response.ok) {
        const errorText = await response.text();
        console.error("Kakao API error response:", errorText);
        return res.status(500).json({ 
          success: false, 
          error: "검색 중 오류가 발생했습니다." 
        });
      }

      const data = await response.json();
      
      const formattedResults = data.documents.map((doc: any) => ({
        id: doc.address?.b_code || doc.road_address?.zone_no || `${doc.x}-${doc.y}`,
        addressName: doc.address_name,
        regionAddress: doc.address ? [
          doc.address.region_1depth_name,
          doc.address.region_2depth_name,
          doc.address.region_3depth_name,
        ].filter(Boolean).join(" ") : null,
        roadAddress: doc.road_address?.address_name || null,
        x: doc.x,
        y: doc.y,
      }));

      res.json({ success: true, data: formattedResults });
    } catch (error) {
      console.error("Kakao API error:", error);
      res.status(500).json({ 
        success: false, 
        error: "네트워크 오류가 발생했습니다." 
      });
    }
  });

  const httpServer = createServer(app);
  
  wsServer.initialize(httpServer);
  
  return httpServer;
}
