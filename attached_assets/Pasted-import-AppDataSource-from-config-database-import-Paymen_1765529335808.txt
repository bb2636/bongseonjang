import { AppDataSource } from "../config/database";
import {
  PaymentTransaction,
  PaymentStatus,
  ProductType,
} from "../entity/PaymentTransaction";
import { MatchingProduct } from "../entity/MatchingProduct";
import { PtProduct } from "../entity/PtProduct";
import { User } from "../entity/User";
import { createHash } from "crypto";
import { NotificationApplicationService } from "./NotificationApplicationService";

const isProduction = process.env.NODE_ENV === 'production';
const NICEPAY_CLIENT_ID = isProduction 
  ? process.env.NICEPAY_PROD_CLIENT_ID 
  : process.env.NICEPAY_CLIENT_ID;
const NICEPAY_SECRET_KEY = isProduction 
  ? process.env.NICEPAY_PROD_SECRET_KEY 
  : process.env.NICEPAY_SECRET_KEY;
const NICEPAY_API_URL = process.env.NICEPAY_API_URL;

console.log(`[NicePay Config] Environment: ${isProduction ? 'production' : 'development'}, Client ID prefix: ${NICEPAY_CLIENT_ID?.substring(0, 3) || 'NOT SET'}`);

interface PaymentParams {
  clientId: string;
  orderId: string;
  amount: number;
  goodsName: string;
  returnUrl: string;
  mallReserved: string;
  buyerName: string;
  buyerEmail: string;
  buyerTel: string;
}

interface CheckoutResponse {
  success: boolean;
  paymentParams?: PaymentParams;
  orderId?: string;
  message?: string;
}

interface PaymentApprovalResponse {
  success: boolean;
  message?: string;
  tid?: string;
  orderId?: string;
  amount?: number;
  approvedAt?: string;
}

interface NicepayCallbackData {
  resultCode: string;
  resultMsg: string;
  tid: string;
  orderId: string;
  amount: string;
  authToken?: string;
  signature?: string;
}

export class PaymentApplicationService {
  private paymentRepository = AppDataSource.getRepository(PaymentTransaction);
  private matchingProductRepository =
    AppDataSource.getRepository(MatchingProduct);
  private ptProductRepository = AppDataSource.getRepository(PtProduct);
  private userRepository = AppDataSource.getRepository(User);
  private notificationService = new NotificationApplicationService();

  private getAuthHeader(): string {
    const credentials = `${NICEPAY_CLIENT_ID}:${NICEPAY_SECRET_KEY}`;
    return `Basic ${Buffer.from(credentials).toString("base64")}`;
  }

  async createMatchingCheckout(
    userId: number,
    productId: number,
    returnUrl: string,
  ): Promise<CheckoutResponse> {
    try {
      console.log("[Payment] createMatchingCheckout called with userId:", userId, "productId:", productId);
      
      const product = await this.matchingProductRepository.findOne({
        where: { id: productId, isActive: true },
      });

      if (!product) {
        console.log("[Payment] Product not found for id:", productId);
        return { success: false, message: "상품을 찾을 수 없습니다." };
      }

      console.log("[Payment] Found product:", product.name);
      
      const user = await this.userRepository.findOne({ where: { id: userId } });
      console.log("[Payment] User query result:", user ? `Found: ${user.nickname}` : "NOT FOUND");
      
      if (!user) {
        console.log("[Payment] User not found for userId:", userId);
        return { success: false, message: "사용자를 찾을 수 없습니다." };
      }

      const timestamp = Date.now();
      const randomStr = Math.random().toString(36).substring(2, 10);
      const orderId = `MATCH-${userId}-${timestamp}-${randomStr}`;

      const transaction = this.paymentRepository.create({
        userId,
        productType: ProductType.MATCHING,
        productId: product.id,
        productName: product.name,
        merchantOrderId: orderId,
        amount: product.price,
        status: PaymentStatus.PENDING,
      });

      await this.paymentRepository.save(transaction);

      const paymentParams = {
        clientId: NICEPAY_CLIENT_ID!,
        orderId: orderId,
        amount: product.price,
        goodsName: product.name,
        returnUrl: returnUrl,
        mallReserved: JSON.stringify({
          transactionId: transaction.id,
          productId: product.id,
          productType: "MATCHING",
        }),
        buyerName: user.nickname || "구매자",
        buyerEmail: user.primaryEmail || "",
        buyerTel: "",
      };

      console.log(
        "[NicePay] Checkout Params:",
        JSON.stringify(paymentParams, null, 2),
      );

      return {
        success: true,
        paymentParams,
        orderId: orderId,
      };
    } catch (error) {
      console.error("Checkout 생성 에러:", error);
      return { success: false, message: "결제 생성 중 오류가 발생했습니다." };
    }
  }

  private verifySignature(
    authToken: string,
    amount: string,
    signature: string,
  ): boolean {
    if (!authToken || !signature || !NICEPAY_CLIENT_ID || !NICEPAY_SECRET_KEY) {
      console.warn("[NicePay] Missing signature verification parameters");
      return false;
    }

    const data = `${authToken}${NICEPAY_CLIENT_ID}${amount}${NICEPAY_SECRET_KEY}`;
    const calculatedSignature = createHash("sha256").update(data).digest("hex");

    const isValid = calculatedSignature === signature;
    console.log("[NicePay] Signature verification:", { isValid });

    return isValid;
  }

  async handleNicepayCallback(
    callbackData: NicepayCallbackData,
  ): Promise<PaymentApprovalResponse> {
    try {
      console.log(
        "[NicePay] Callback received:",
        JSON.stringify(callbackData, null, 2),
      );

      const {
        resultCode,
        resultMsg,
        tid,
        orderId,
        amount,
        authToken,
        signature,
      } = callbackData;

      const transaction = await this.paymentRepository.findOne({
        where: { merchantOrderId: orderId },
      });

      if (!transaction) {
        console.error("[NicePay] Transaction not found for orderId:", orderId);
        return { success: false, message: "결제 정보를 찾을 수 없습니다." };
      }

      if (transaction.status === PaymentStatus.PAID) {
        console.log("[NicePay] Transaction already paid:", orderId);
        return {
          success: true,
          tid,
          orderId,
          amount: transaction.amount,
          approvedAt: transaction.paidAt?.toISOString(),
        };
      }

      if (resultCode !== "0000") {
        await this.paymentRepository.update(transaction.id, {
          status: PaymentStatus.FAILED,
          pgResponseCode: resultCode,
          pgResponseMessage: resultMsg,
        });
        return {
          success: false,
          message: resultMsg || "결제 인증에 실패했습니다.",
        };
      }

      if (authToken && signature) {
        const isValidSignature = this.verifySignature(
          authToken,
          amount,
          signature,
        );
        if (!isValidSignature) {
          console.error(
            "[NicePay] Signature verification failed for orderId:",
            orderId,
          );
          await this.paymentRepository.update(transaction.id, {
            status: PaymentStatus.FAILED,
            pgResponseCode: "INVALID_SIGNATURE",
            pgResponseMessage: "결제 서명 검증에 실패했습니다.",
          });
          return { success: false, message: "결제 서명 검증에 실패했습니다." };
        }
      }

      const numericAmount = Number(amount);
      if (transaction.amount !== numericAmount) {
        console.error("[NicePay] Amount mismatch:", {
          expected: transaction.amount,
          received: numericAmount,
        });
        await this.paymentRepository.update(transaction.id, {
          status: PaymentStatus.FAILED,
          pgResponseCode: "AMOUNT_MISMATCH",
          pgResponseMessage: "결제 금액이 일치하지 않습니다.",
        });
        return { success: false, message: "결제 금액이 일치하지 않습니다." };
      }

      console.log("[NicePay] Calling approval API for tid:", tid);

      const approvalResponse = await fetch(
        `${NICEPAY_API_URL}/payments/${tid}`,
        {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization: this.getAuthHeader(),
          },
          body: JSON.stringify({ amount: numericAmount }),
        },
      );

      const approvalResult = await approvalResponse.json();
      console.log(
        "[NicePay] Approval Response:",
        JSON.stringify(approvalResult, null, 2),
      );

      if (approvalResult.resultCode === "0000") {
        await this.paymentRepository.update(transaction.id, {
          status: PaymentStatus.PAID,
          nicepayTid: tid,
          paymentMethod: approvalResult.payMethod || "CARD",
          approvalNumber: approvalResult.approvalNo,
          pgResponseCode: approvalResult.resultCode,
          pgResponseMessage: approvalResult.resultMsg,
          paidAt: new Date(),
        });

        const updatedTransaction = await this.paymentRepository.findOne({
          where: { id: transaction.id },
        });

        if (updatedTransaction) {
          let ticketQuantity = 1;
          if (updatedTransaction.productType === ProductType.MATCHING) {
            const product = await this.matchingProductRepository.findOne({
              where: { id: updatedTransaction.productId },
            });
            if (product && product.type !== 'UNLIMITED') {
              ticketQuantity = product.ticketQuantity || 1;
            }
          }
          
          await this.grantProductBenefits(updatedTransaction);
          
          try {
            await this.notificationService.createPaymentNotification(
              updatedTransaction.userId,
              updatedTransaction.productName,
              updatedTransaction.amount || 0,
              updatedTransaction.merchantOrderId,
              ticketQuantity,
              updatedTransaction.productType as 'MATCHING' | 'PT'
            );
          } catch (notifError) {
            console.error("[NicePay] Notification creation error:", notifError);
          }
        }

        return {
          success: true,
          tid,
          orderId,
          amount: numericAmount,
          approvedAt: new Date().toISOString(),
        };
      } else {
        await this.paymentRepository.update(transaction.id, {
          status: PaymentStatus.FAILED,
          pgResponseCode: approvalResult.resultCode,
          pgResponseMessage: approvalResult.resultMsg,
        });
        return {
          success: false,
          message: approvalResult.resultMsg || "결제 승인에 실패했습니다.",
        };
      }
    } catch (error) {
      console.error("[NicePay] Callback processing error:", error);
      return { success: false, message: "결제 처리 중 오류가 발생했습니다." };
    }
  }

  private async grantProductBenefits(
    transaction: PaymentTransaction,
  ): Promise<void> {
    try {
      const user = await this.userRepository.findOne({
        where: { id: transaction.userId },
      });
      if (!user) return;

      if (transaction.productType === ProductType.MATCHING) {
        const product = await this.matchingProductRepository.findOne({
          where: { id: transaction.productId },
        });

        if (product) {
          const bonusPoints = product.bonusPoints || 0;
          user.point = (user.point || 0) + bonusPoints;

          if (product.type === "UNLIMITED") {
            const validDays = product.validDays || 30;
            const paidAt = transaction.paidAt || new Date();
            const now = new Date();
            
            const currentSubscription = await this.paymentRepository
              .createQueryBuilder('payment')
              .where('payment.userId = :userId', { userId: user.id })
              .andWhere('payment.id != :currentId', { currentId: transaction.id })
              .andWhere('payment.status = :status', { status: PaymentStatus.PAID })
              .andWhere('payment.subscriptionExpireAt IS NOT NULL')
              .andWhere('payment.subscriptionExpireAt > :now', { now })
              .orderBy('payment.subscriptionExpireAt', 'DESC')
              .getOne();
            
            const baseDate = currentSubscription?.subscriptionExpireAt && currentSubscription.subscriptionExpireAt > paidAt
              ? currentSubscription.subscriptionExpireAt
              : paidAt;
            
            const expireAt = new Date(baseDate);
            expireAt.setDate(expireAt.getDate() + validDays);
            
            await this.paymentRepository.update(transaction.id, {
              subscriptionExpireAt: expireAt
            });
            
            console.log("[NicePay] Subscription granted:", {
              userId: user.id,
              productId: product.id,
              validDays,
              baseDate,
              expireAt,
              points: bonusPoints,
            });
          } else {
            const ticketQuantity = product.ticketQuantity;
            user.matchingTicket = (user.matchingTicket || 0) + ticketQuantity;
            
            console.log("[NicePay] Benefits granted:", {
              userId: user.id,
              tickets: ticketQuantity,
              points: bonusPoints,
            });
          }

          await this.userRepository.save(user);
        }
      } else if (transaction.productType === ProductType.PT) {
        const product = await this.ptProductRepository.findOne({
          where: { id: transaction.productId },
        });

        if (product) {
          const exposureDays = product.exposureDays || 30;
          const paidAt = transaction.paidAt || new Date();
          const now = new Date();
          
          const currentSubscription = await this.paymentRepository
            .createQueryBuilder('payment')
            .where('payment.userId = :userId', { userId: user.id })
            .andWhere('payment.id != :currentId', { currentId: transaction.id })
            .andWhere('payment.productType = :productType', { productType: ProductType.PT })
            .andWhere('payment.status = :status', { status: PaymentStatus.PAID })
            .andWhere('payment.subscriptionExpireAt IS NOT NULL')
            .andWhere('payment.subscriptionExpireAt > :now', { now })
            .orderBy('payment.subscriptionExpireAt', 'DESC')
            .getOne();
          
          const baseDate = currentSubscription?.subscriptionExpireAt && currentSubscription.subscriptionExpireAt > paidAt
            ? currentSubscription.subscriptionExpireAt
            : paidAt;
          
          const expireAt = new Date(baseDate);
          expireAt.setDate(expireAt.getDate() + exposureDays);
          
          await this.paymentRepository.update(transaction.id, {
            subscriptionExpireAt: expireAt
          });
          
          console.log("[NicePay] PT Subscription granted:", {
            userId: user.id,
            productId: product.id,
            exposureDays,
            baseDate,
            expireAt,
          });
        }
      }
    } catch (error) {
      console.error("상품 혜택 지급 에러:", error);
    }
  }

  async getPaymentByOrderId(
    orderId: string,
  ): Promise<PaymentTransaction | null> {
    return this.paymentRepository.findOne({
      where: { merchantOrderId: orderId },
    });
  }

  async getPaymentHistory(userId: number): Promise<PaymentTransaction[]> {
    return this.paymentRepository.find({
      where: { userId },
      order: { createdAt: "DESC" },
    });
  }
}
