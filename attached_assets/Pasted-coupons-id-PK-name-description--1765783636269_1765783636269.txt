coupons { // 쿠폰(공통/마스터)
  id;                 // PK
  name;               // 쿠폰명
  description;        // 쿠폰 설명(노출용, 선택)
  min_order_amount;   // 최소 주문 금액
  is_active;          // 활성/비활성
  created_at;         // 생성일
  updated_at;         // 수정일
}

coupon_fixed_discounts { // 정액 할인 정책
  coupon_id;         // PK, FK -> coupons.id
  discount_amount;   // 할인 금액(원)
}

coupon_rate_discounts { // 정율 할인 정책
  coupon_id;            // PK, FK -> coupons.id
  discount_rate;        // 할인율(%)
  max_discount_amount;  // 최대 할인 금액(원)
}

coupon_shipping_discounts { // 배송비 할인 정책
  coupon_id;                // PK, FK -> coupons.id
  is_free_shipping;         // 무료배송 여부 (보통 true)
  shipping_discount_amount; // 배송비 할인 금액(원, 무료배송이면 0 가능)
}

coupon_validity_fixed_ranges { // 기간 정책: 고정기간(시작~종료)
  coupon_id;  // PK, FK -> coupons.id
  start_at;   // 사용 시작일시
  end_at;     // 사용 종료일시
}

coupon_validity_after_issue_days { // 기간 정책: 발급일 기준 N일
  coupon_id;   // PK, FK -> coupons.id
  valid_days;  // 발급 후 유효 일수 (예: 7)
}

coupon_apply_all_products { // 적용대상: 전체상품
  coupon_id;  // PK, FK -> coupons.id
}

coupon_apply_categories { // 적용대상: 특정카테고리(매핑)
  coupon_id;    // PK(복합), FK -> coupons.id
  category_id;  // PK(복합), FK -> categories.id
}

coupon_issue_all_users { // 발급대상: 전체회원
  coupon_id;  // PK, FK -> coupons.id
}

coupon_issue_new_users { // 발급대상: 신규가입회원
  coupon_id;  // PK, FK -> coupons.id
}

coupon_issue_grades { // 발급대상: 특정등급(매핑)
  coupon_id;  // PK(복합), FK -> coupons.id
  grade_id;   // PK(복합), FK -> user_grades.id
}

user_coupons { // 유저 쿠폰(발급/사용 이력: 실제 쿠폰 1장)
  id;            // PK
  coupon_id;     // FK -> coupons.id
  user_id;       // FK -> users.id

  status;        // ISSUED / USED / EXPIRED / CANCELLED
  issued_at;     // 발급일시

  valid_from;    // 실제 사용 시작일시(발급 시 확정)
  valid_to;      // 실제 사용 종료일시(발급 시 확정)

  used_at;       // 사용일시(선택)
  used_order_id; // FK -> orders.id (선택)

  created_at;    // 생성일(선택: issued_at로 대체 가능)
  updated_at;    // 수정일(선택)
}
